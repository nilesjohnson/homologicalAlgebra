\begin{theorem}
  Let $P$ be a flat complex with each $d(P_n)$ flat.  Then for all

  \[
  0 \rightarrow H_n(P) \otimes M 
  \rightarrow H_n(P \otimes M)
  \rightarrow Tor_1(H_{n-1}(P), M) \rightarrow 0
